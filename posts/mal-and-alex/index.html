mps <- soccer %>% 
  group_by(Player, Squad) %>% 
  summarise(
    TotalGames = sum(MP), 
    TotalStarts = sum(Starts), 
    Total90s = sum(`90s`))%>% 
  mutate(
    TotalMatches = TotalGames, 
    TotalStarts = TotalStarts)
